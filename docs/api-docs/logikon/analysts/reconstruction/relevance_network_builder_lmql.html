257    """RelevanceNetworkBuilderLMQL
258
259    This LMQLAnalyst is responsible for creating a relevance network,
260    i.e. a (quasi-) complete fuzzy argmap, from a pros and cons list and a given issue.

275    def _unpack_reason(self, reason_data: dict, issue: str) -> list[Claim]:
276        """Internal (class method) wrapper for lmql query function."""
277        return unpack_reason(
278            reason_data=reason_data,
279            issue=issue,
280            prmpt_data=self._prompt_template.to_dict(),
281            model=self._model,
282            **self._generation_kwargs,
283        )

285    def _unpack_reasons(
286        self, pros_and_cons: ProsConsList, issue: str
287    ) -> tuple[ProsConsList, list[tuple[dict, list[dict]]]]:
288        """Unpacks each individual reason in a pros and cons list
289
290        Args:
291            pros_and_cons (ProsConsList): pros and cons list with reason to be unpacked
292            issue (str): overarching issue addressed by pros and cons
293
294        Returns:
295            ProsConsList, unpacking: pros and cons list with unpacked reasons; list of tuples with original and unpacked claims
296        """
297
298        pros_and_cons = copy.deepcopy(pros_and_cons)
299        unpacking = []
300
301        for root in pros_and_cons.roots:
302            to_be_added = []
303            to_be_removed = []
304            for pro in root.pros:
305                unpacked_pros = self._unpack_reason(reason_data=pro.dict(), issue=issue)
306                if len(unpacked_pros) > 1:
307                    unpacking.append((pro.dict(), [claim.dict() for claim in unpacked_pros]))
308                    to_be_removed.append(pro)
309                    to_be_added.extend(unpacked_pros)
310            for pro in to_be_removed:
311                root.pros.remove(pro)
312            root.pros.extend(to_be_added)
313            to_be_added = []
314            to_be_removed = []
315            for con in root.cons:
316                unpacked_cons = self._unpack_reason(reason_data=con.dict(), issue=issue)
317                if len(unpacked_cons) > 1:
318                    unpacking.append((con.dict(), [claim.dict() for claim in unpacked_cons]))
319                    to_be_removed.append(con)
320                    to_be_added.extend(unpacked_cons)
321            for con in to_be_removed:
322                root.cons.remove(con)
323            root.cons.extend(to_be_added)
324
325        return pros_and_cons, unpacking

327    # the following function calculates the strength of the argumentative relation between tow claims
328    def _relation_strength(
329        self, source_node: ArgMapNode, target_node: ArgMapNode, valence: str | None = None, issue: str = ""
330    ) -> tuple[float, str]:
331        """Calculate the valence and strength of an argumentative relation
332
333        Calculate the valence and strength of the argumentative relation between two
334        argument map nodes (from source to target).
335
336        Args:
337            target_node (ArgMapNode): target
338            source_node (ArgMapNode): source
339            valence (str, optional): calculate strength between source and target for this valence
340
341        Returns:
342            Tuple[float, str]: strength, valence
343        """
344
345        # step 1: valence probs
346        lmql_result = lmql_queries.valence(
347            {"label": source_node.label, "text": source_node.text},
348            {"label": target_node.label, "text": target_node.text},
349            issue=issue,
350            prmpt_data=self._prompt_template.to_dict(),
351            model=self._model,
352            **self._generation_kwargs,
353        )
354        # self.logger.debug(f"Eliciting valence with lmql query: '{lmql_result.prompt}'")
355        if valence is None:
356            valence = lmql_queries.label_to_valence(lmql_result.variables[lmql_result.distribution_variable])
357        prob_1 = next(
358            prob
359            for label, prob in lmql_queries.get_distribution(lmql_result)
360            if lmql_queries.label_to_valence(label) == valence
361        )
362
363        # step 2: strength probs
364        query_fn = lmql_queries.supports_q if valence == am.SUPPORT else lmql_queries.attacks_q
365        lmql_result = query_fn(
366            {"label": source_node.label, "text": source_node.text},
367            {"label": target_node.label, "text": target_node.text},
368            prmpt_data=self._prompt_template.to_dict(),
369            model=self._model,
370            **self._generation_kwargs,
371        )
372        prob_2 = next(prob for label, prob in lmql_queries.get_distribution(lmql_result) if label == "A")
373
374        return prob_1 * prob_2, valence

376    def _dialectically_equivalent(self, relevance_network: FuzzyArgMap, node1: ArgMapNode, node2: ArgMapNode) -> bool:
377        """Checks whether two reasons are dialectically equivalent relative to central claims
378
379        If two reasons have common parent central claims, they are dialectically equivalent
380        iff they have identical argumentative relations to each common root claim.
381        If two reasons have no common root claims, they are dialectically equivalent
382        iff one is only supporting and the other only attacking central claims.
383
384        This function is hence assuming that the central claims are mutually exclusive (contrary).
385
386        Args:
387            relevance_network (FuzzyArgMap): argmap relative to which dialectic equivalence is assessed
388            source_node (ArgMapNode): first node
389            target_node (ArgMapNode): second node
390
391        Returns:
392            bool: true if two nodes are dialectically equivalent, false otherwise
393        """
394        # determine parent central claims of two nodes
395        node1_parents = {
396            edge.target
397            for edge in relevance_network.edgelist
398            if edge.source == node1.id and relevance_network.get_node_type(edge.target) == am.CENTRAL_CLAIM
399        }
400        node2_parents = {
401            edge.target
402            for edge in relevance_network.edgelist
403            if edge.source == node2.id and relevance_network.get_node_type(edge.target) == am.CENTRAL_CLAIM
404        }
405
406        common_parents = node1_parents.intersection(node2_parents)
407
408        # Case 1: there are common parents
409        if len(common_parents) > 0:
410            equivalent = True
411            for parent in common_parents:
412                node1_valence = next(
413                    edge.valence
414                    for edge in relevance_network.edgelist
415                    if edge.source == node1.id and edge.target == parent
416                )
417                node2_valence = next(
418                    edge.valence
419                    for edge in relevance_network.edgelist
420                    if edge.source == node2.id and edge.target == parent
421                )
422                if node1_valence != node2_valence:
423                    equivalent = False
424                    break
425            return equivalent
426
427        # Case 2: there are no common parents
428        node1_valences = {
429            edge.valence
430            for edge in relevance_network.edgelist
431            if edge.source == node1.id and edge.target in node1_parents
432        }
433        node2_valences = {
434            edge.valence
435            for edge in relevance_network.edgelist
436            if edge.source == node2.id and edge.target in node2_parents
437        }
438        equivalent = len(node1_valences.intersection(node2_valences)) == 0
439
440        return equivalent

442    def _add_node(self, argmap: FuzzyArgMap, claim: Claim, node_type: str = am.REASON) -> ArgMapNode:
443        """Add node to fuzzy argmap
444
445        Args:
446            argmap (FuzzyArgMap): fuzzy argmap
447            claim (Claim): claim to be added
448
449        Returns:
450            ArgMapNode: node added to fuzzy argmap
451        """
452        node = ArgMapNode(
453            id=str(uuid.uuid4()),
454            label=claim.label,
455            text=claim.text,
456            node_type=node_type,
457        )
458        argmap.nodelist.append(node)
459        return node

461    def _add_fuzzy_edge(
462        self,
463        argmap: FuzzyArgMap,
464        source_node: ArgMapNode,
465        target_node: ArgMapNode,
466        valence: str | None = None,
467        issue: str = "",
468    ) -> FuzzyArgMapEdge:
469        """Add fuzzy edge to fuzzy argmap
470
471        Args:
472            argmap (FuzzyArgMap): fuzzy argmap
473            target_node (ArgMapNode): target node
474            source_node (ArgMapNode): source node
475            valence (str, optional): fixed valence to assume, automatically determines most likely val if None
476            issue (str, optional): issue addressed by arguments
477
478        Returns:
479            FuzzyArgMapEdge: newly added edge
480        """
481        w, val = self._relation_strength(source_node, target_node, valence=valence, issue=issue)
482        if valence is not None:
483            val = valence
484        edge = FuzzyArgMapEdge(source=source_node.id, target=target_node.id, valence=val, weight=w)
485        argmap.edgelist.append(edge)
486        return edge

488    def _analyze(self, analysis_state: AnalysisState):
489        """Build fuzzy argmap from pros and cons.
490
491        Args:
492            analysis_state (AnalysisState): current analysis_state to which new artifact is added
493
494        Raises:
495            ValueError: Failure to create Fuzzy argument map
496
497        Proceeds as follows:
498
499        1. Unpack each individual pro / con in pros cons list into separate claims (if possible)
500        2. Determine reason - root weights for all reasons subsumed under a root claim
501        3. Determine reason - reason weights for all reason pairs
502
503        """
504
505        issue = next((a.data for a in analysis_state.artifacts if a.id == "issue"), None)
506        if issue is None:
507            msg = f"Missing required artifact: issue. Available artifacts: {analysis_state.artifacts!s}"
508            raise ValueError(msg)
509
510        pros_and_cons_data = next((a.data for a in analysis_state.artifacts if a.id == "proscons"), None)
511        if pros_and_cons_data is None:
512            msg = f"Missing required artifact: proscons. Available artifacts: {analysis_state.artifacts!s}"
513            raise ValueError(msg)
514        try:
515            pros_and_cons = ProsConsList(**pros_and_cons_data)
516        except Exception as err:
517            msg = f"Missing required artifact: issue. Available artifacts: {analysis_state.artifacts!s}"
518            raise ValueError(msg) from err
519
520        # unpack individual reasons
521        pros_and_cons, unpacking = self._unpack_reasons(pros_and_cons, issue)
522        self.logger.debug(f"Unpacked pros and cons list: {pprint.pformat(pros_and_cons.dict())}")
523
524        # create fuzzy argmap from fuzzy pros and cons list (reason-root edges)
525        relevance_network = FuzzyArgMap()
526        for root in tqdm.tqdm(pros_and_cons.roots, desc="Reason-root edges"):
527            target_node = self._add_node(relevance_network, root, node_type=am.CENTRAL_CLAIM)
528            for pro in root.pros:
529                source_node = self._add_node(relevance_network, pro, node_type=am.REASON)
530                self._add_fuzzy_edge(
531                    relevance_network, source_node=source_node, target_node=target_node, valence=am.SUPPORT, issue=issue
532                )
533            for con in root.cons:
534                source_node = self._add_node(relevance_network, con, node_type=am.REASON)
535                self._add_fuzzy_edge(
536                    relevance_network, source_node=source_node, target_node=target_node, valence=am.ATTACK, issue=issue
537                )
538
539        # TODO: improve naive sampling
540        # add fuzzy reason-reason edges
541        for target_node in tqdm.tqdm(relevance_network.nodelist, desc="Reason-reason edges"):
542            if target_node.node_type == am.CENTRAL_CLAIM:
543                continue
544
545            source_nodes = [
546                node for node in relevance_network.nodelist if node.id != target_node.id and node.node_type == am.REASON
547            ]
548            if len(source_nodes) > MAX_N_RELATIONS:
549                source_nodes = random.sample(source_nodes, MAX_N_RELATIONS)
550
551            for source_node in source_nodes:
552                valence = None
553                if self._keep_pcl_valences:
554                    valence = (
555                        am.SUPPORT
556                        if self._dialectically_equivalent(relevance_network, source_node, target_node)
557                        else am.ATTACK
558                    )
559                self._add_fuzzy_edge(
560                    relevance_network, source_node=source_node, target_node=target_node, valence=valence
561                )
562
563        if relevance_network is None:
564            self.logger.warning("Failed to build relevance network (relevance_network is None).")
565
566        try:
567            relevance_network_data = relevance_network.model_dump()  # type: ignore
568        except AttributeError:
569            relevance_network_data = relevance_network.dict()
570
571        artifact = Artifact(
572            id=self.get_product(),
573            description=self.get_description(),
574            data=relevance_network_data,
575            metadata={"unpacking": unpacking},
576        )
577
578        analysis_state.artifacts.append(artifact)

377    def _dialectically_equivalent(self, relevance_network: FuzzyArgMap, node1: ArgMapNode, node2: ArgMapNode) -> bool:
378        """Checks whether two reasons are dialectically equivalent relative to central claims
379
380        If two reasons have common parent central claims, they are dialectically equivalent
381        iff they have identical argumentative relations to each common root claim.
382        If two reasons have no common root claims, they are dialectically equivalent
383        iff one is only supporting and the other only attacking central claims.
384
385        This function is hence assuming that the central claims are mutually exclusive (contrary).
386
387        Args:
388            relevance_network (FuzzyArgMap): argmap relative to which dialectic equivalence is assessed
389            source_node (ArgMapNode): first node
390            target_node (ArgMapNode): second node
391
392        Returns:
393            bool: true if two nodes are dialectically equivalent, false otherwise
394        """
395        # determine parent central claims of two nodes
396        node1_parents = {
397            edge.target
398            for edge in relevance_network.edgelist
399            if edge.source == node1.id and relevance_network.get_node_type(edge.target) == am.CENTRAL_CLAIM
400        }
401        node2_parents = {
402            edge.target
403            for edge in relevance_network.edgelist
404            if edge.source == node2.id and relevance_network.get_node_type(edge.target) == am.CENTRAL_CLAIM
405        }
406
407        common_parents = node1_parents.intersection(node2_parents)
408
409        # Case 1: there are common parents
410        if len(common_parents) > 0:
411            equivalent = True
412            for parent in common_parents:
413                node1_valence = next(
414                    edge.valence
415                    for edge in relevance_network.edgelist
416                    if edge.source == node1.id and edge.target == parent
417                )
418                node2_valence = next(
419                    edge.valence
420                    for edge in relevance_network.edgelist
421                    if edge.source == node2.id and edge.target == parent
422                )
423                if node1_valence != node2_valence:
424                    equivalent = False
425                    break
426            return equivalent
427
428        # Case 2: there are no common parents
429        node1_valences = {
430            edge.valence
431            for edge in relevance_network.edgelist
432            if edge.source == node1.id and edge.target in node1_parents
433        }
434        node2_valences = {
435            edge.valence
436            for edge in relevance_network.edgelist
437            if edge.source == node2.id and edge.target in node2_parents
438        }
439        equivalent = len(node1_valences.intersection(node2_valences)) == 0
440
441        return equivalent
