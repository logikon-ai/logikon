397    def parse_list(inputs: dict[str, str]) -> dict[str, list[str]]:
398        list_items: list[str] = []
399        list_text = inputs.get("list_text", "")
400        text = list_text.strip(" \n")
401        for line in text.split("\n"):
402            line = line.strip()
403            if line[:2] in ["1.", "2.", "3.", "4.", "5.", "6.", "7.", "8.", "9."]:
404                line = line[2:]
405                line = line.strip()
406                if line:
407                    list_items.append(line)
408        return {"list_items": list_items}

410    def _init_argmap(self, claims) -> InformalArgMap:
411        """
412        initializes argmap with central claims (mutually exclusive)
413        """
414        argmap = InformalArgMap()
415        for e, claim in enumerate(claims):
416            argmap.nodelist.append(
417                ArgMapNode(
418                    id=str(uuid.uuid4()),
419                    text=claim,
420                    label=f"Claim-{e+1}",
421                    nodeType="proposition",
422                    annotationReferences=[],
423                )
424            )
425        for claim_a in argmap.nodelist:
426            for claim_b in argmap.nodelist:
427                if claim_a != claim_b:
428                    argmap.edgelist.append(
429                        ArgMapEdge(
430                            source=claim_a.id,
431                            target=claim_b.id,
432                            valence=am.ATTACK,
433                        )
434                    )
435        return argmap

437    def _is_pro_reason(self, claim: str, reason: str) -> bool:
438        if isinstance(self.llm, OpenAI):
439            llm_kwargs = {"max_tokens": 4, "temperature": 0.4}
440        else:
441            llm_kwargs = {}
442        chain_q_supports = LLMChain(
443            llm=self.llm, prompt=self.prompt_registry["prompt_q_supports"], verbose=self.verbose, llm_kwargs=llm_kwargs
444        )
445        answer = chain_q_supports.run(premise=reason, hypothesis=claim)
446        print(f"> Answer: {answer}")
447        answer = answer.strip(" \n")
448        is_pro = (
449            answer.lower().startswith("b")
450            or "b)" in answer.lower()
451            and "a)" not in answer.lower()
452            and "c)" not in answer.lower()
453        )
454        return is_pro

508        """checks if two reasons are equivalent"""
509        if isinstance(self.llm, OpenAI):

516            verbose=self.verbose,

520        answer = str(answer)
521        print(f"> Answer: {answer}")
522        answer = answer.strip(" \n")

524        answer = answer.strip("\" ")

531    def _is_redundant(self, argmap: InformalArgMap, reason: str, annotations: List[AnnotationSpan]) -> bool:
532        """
533        checks if reason is redundant, i.e. equivalent to a reason already in the argmap
534        """
535        if not reason:
536            return True
537
538        # check redundancy only for nodes with overlapping annotation span
539        overlapping_nodes = []
540        for node in argmap.nodelist:
541            overlaps = False
542            for span1 in annotations:
543                for span2 in node.annotations:
544                    if span1.start < span2.end and span2.start < span1.end:
545                        overlaps = True
546                        break
547            if overlaps:
548                overlapping_nodes.append(node)
549
550        for node in overlapping_nodes:
551            if self._are_equivalent(reason, node.text):
552                return True
553
554        return False

556    def _shorten_reason(self, reason: str, claim: str = "", valence: str = "for") -> str:
557        """
558        subcall: shorten reason if necessary
559        """
560
561        if not reason:
562            return reason
563        print(f"Word count current reason: {len(reason.split(' '))}")
564
565        chain_shorten = LLMChain(llm=self.llm, prompt=self.prompt_registry["prompt_shorten"], verbose=self.verbose)
566        if len(reason.split(" ")) > self.max_words_reason:
567            gist = chain_shorten.run(reason=reason, valence=valence, claim=claim)
568            gist = gist.strip(" \n")
569            print(f"Word count shortened gist: {len(gist.split(' '))}")
570            reason = gist if len(gist.split(" ")) < len(reason.split(" ")) else reason
571
572        return reason

574    def _process_reasons(self, reasons: str, claim: str = "", valence: str = "for") -> list[str]:
575        """
576        subcall: splits reasons list into individual reasons and shortnes each if necessary
577        """
578
579        parse_chain = TransformChain(input_variables=["list_text"], output_variables=["list_items"], transform=self.parse_list)  # type: ignore
580        reasons_l = parse_chain.run({"list_text": reasons})
581        reasons_l = [self._shorten_reason(reason, valence=valence, claim=claim) for reason in reasons_l if reason]
582        return reasons_l

584    def _add_argument(
585        self,
586        argmap: InformalArgMap,
587        reason: str,
588        label: str = "",
589        target_id: Optional[str] = None,
590        valence: str = "for",
591        annotations: Optional[list[AnnotationSpan]] = None,
592    ) -> ArgMapNode:
593        """
594        subcall: adds and returns node
595        """
596        if annotations is None:
597            annotations = []
598        node_id = str(uuid.uuid4())
599        node = ArgMapNode(
600            id=node_id,
601            text=reason,
602            label=label,
603            nodeType="proposition",
604            annotations=annotations,
605        )
606        argmap.nodelist.append(node)
607
608        if target_id:
609            edge = ArgMapEdge(
610                source=node.id,
611                target=target_id,
612                valence=am.SUPPORT if valence == "for" else am.ATTACK,
613            )
614            argmap.edgelist.append(edge)
615
616        return node

618    def _match_quote(self, quote: str, source_text: str) -> list[AnnotationSpan]:
619        """
620        utility function:
621        tries to match quote to a text spans
622        returns list of text spans
623        """
624        quote = quote.strip(' "\n')
625        if not quote:
626            return []
627
628        # normalize quote
629        quote = quote.lower()
630        quote = re.sub(r"\W+", "", quote)
631
632        # normalize source text
633        source_text = source_text.lower()
634        source_text = re.sub(r"\W+", "", source_text)
635
636        annotations: list[AnnotationSpan] = []
637
638        while quote in source_text:
639            start = source_text.find(quote)
640            end = start + len(quote)
641            annotations.append(AnnotationSpan(start=start, end=end))
642            source_text = source_text[end:]
643
644        return annotations

646    def _process_and_add_arguments(
647        self, argmap: InformalArgMap, completion: str, reasons: str, target_node: ArgMapNode, valence: str = "for"
648    ) -> list[ArgMapNode]:
649        """
650        subcall:
651        - splits and processes reasons,
652        - adds arguments and edges to argmap,
653        - returns newly added nodes
654        """
655        new_nodes = []
656        claim = target_node.text
657        reasons_l = self._process_reasons(reasons, claim=claim, valence=valence)
658
659        doublecheck = self._is_pro_reason if valence == "for" else self._is_con_reason
660        reasons_l = [reason for reason in reasons_l if doublecheck(claim=claim, reason=reason)]
661        reasons_l = reasons_l[: self.max_parallel_reasons]  # cut off reasons
662
663        chain_headline = LLMChain(llm=self.llm, prompt=self.prompt_registry["prompt_headline"], verbose=self.verbose)
664        chain_annotation = LLMChain(
665            llm=self.llm, prompt=self.prompt_registry["prompt_annotation"], verbose=self.verbose
666        )
667
668        for reason in reasons_l:
669            headline: str = chain_headline.run(reason=reason, claim=claim, valence=valence)
670            headline = headline.strip(" \n")
671            headline = headline.split("\n")[0]
672            headline = " ".join(headline.split()[: self.max_words_title])
673            quote = chain_annotation.run(source_text=completion, claim=reason)
674            print(f"> Answer: {quote}")
675            annotations = self._match_quote(quote, completion)
676            # only add argument if a match has been found
677            if annotations:
678                if not self._is_redundant(argmap=argmap, reason=reason, annotations=annotations):
679                    new_node = self._add_argument(
680                        argmap=argmap,
681                        reason=reason,
682                        label=headline,
683                        target_id=target_node.id,
684                        valence=valence,
685                        annotations=annotations,
686                    )
687                    new_nodes.append(new_node)
688
689        return new_nodes

691    def _call(
692        self, inputs: dict[str, Any], run_manager: Optional[CallbackManagerForChainRun] = None
693    ) -> dict[str, dict]:
694        # define subchains
695        chain_pros = LLMChain(llm=self.llm, prompt=self.prompt_registry["prompt_pros"], verbose=self.verbose)
696        chain_cons = LLMChain(llm=self.llm, prompt=self.prompt_registry["prompt_cons"], verbose=self.verbose)
697
698        completion = inputs["completion"]
699        claims = inputs["claims"]
700        argmap = self._init_argmap(claims)
701
702        new_nodes: list[ArgMapNode] = argmap.nodelist
703
704        for depth in range(self.argmap_depth):
705            target_nodes = copy.deepcopy(new_nodes)
706            new_nodes = []
707
708            print("###")
709            print(f"Mining arguments at depth {depth+1}.")
710            print(f" Number of target nodes: {len(target_nodes)}.")
711            print(f" Number of nodes in argmap: {len(argmap.nodelist)}.")
712            print("###")
713
714            for enum, target_node in enumerate(target_nodes):
715                print(f"### Processing target node {enum} of {len(target_nodes)} at depth {depth+1}. ###")
716                claim = target_node.text
717
718                is_supported = self._is_supported(claim=claim, source_text=completion)
719                is_attacked = self._is_attacked(claim=claim, source_text=completion)
720
721                if is_supported:
722                    pros = chain_pros.run(claim=claim, source_text=completion)
723                    print(f"> Answer: {pros}")
724                    new_pros = self._process_and_add_arguments(
725                        argmap=argmap, completion=completion, reasons=pros, target_node=target_node, valence="for"
726                    )
727                    new_nodes.extend(new_pros)
728

731                    print(f"> Answer: {cons}")

733                        argmap=argmap, completion=completion, reasons=cons, target_node=target_node, valence="against"
734                    )
735                    new_nodes.extend(new_cons)

738            argmap_dict = argmap.model_dump()  # type: ignore
739        except AttributeError:

398    def parse_list(inputs: dict[str, str]) -> dict[str, list[str]]:
399        list_items: list[str] = []
400        list_text = inputs.get("list_text", "")
401        text = list_text.strip(" \n")
402        for line in text.split("\n"):
403            line = line.strip()
404            if line[:2] in ["1.", "2.", "3.", "4.", "5.", "6.", "7.", "8.", "9."]:
405                line = line[2:]
406                line = line.strip()
407                if line:
408                    list_items.append(line)

411    def _init_argmap(self, claims) -> InformalArgMap:
412        """
413        initializes argmap with central claims (mutually exclusive)
414        """
415        argmap = InformalArgMap()
416        for e, claim in enumerate(claims):
417            argmap.nodelist.append(
418                ArgMapNode(
419                    id=str(uuid.uuid4()),
420                    text=claim,
421                    label=f"Claim-{e+1}",
422                    nodeType="proposition",
423                    annotationReferences=[],
424                )
425            )
426        for claim_a in argmap.nodelist:
427            for claim_b in argmap.nodelist:
428                if claim_a != claim_b:
429                    argmap.edgelist.append(
430                        ArgMapEdge(
431                            source=claim_a.id,
432                            target=claim_b.id,
433                            valence=am.ATTACK,
434                        )
435                    )
436        return argmap

438    def _is_pro_reason(self, claim: str, reason: str) -> bool:
439        if isinstance(self.llm, OpenAI):
440            llm_kwargs = {"max_tokens": 4, "temperature": 0.4}
441        else:
442            llm_kwargs = {}

445        )

447        print(f"> Answer: {answer}")

449        is_pro = (

453            and "c)" not in answer.lower()

Abstract base class for creating structured sequences of calls to components.

Chains should be used to encode a sequence of calls to components like models, document retrievers, other chains, etc., and provide a simple interface to this sequence.

The Chain interface makes it easy to create apps that are: - Stateful: add Memory to any Chain to give it state, - Observable: pass Callbacks to a Chain to execute additional functionality, like logging, outside the main sequence of component calls, - Composable: the Chain API is flexible enough that it is easy to combine Chains with other components, including other Chains.

The main methods exposed by chains are: - __call__: Chains are callable. The __call__ method is the primary way to execute a Chain. This takes inputs as a dictionary and returns a dictionary output. - run: A convenience method that takes inputs as args/kwargs and returns the output as a string or object. This method can only be used for a subset of chains and cannot return as rich of an output as __call__.
